CREATE TABLE `bulbow`.`comments` ( `com_id` INT(11) NOT NULL AUTO_INCREMENT , `post_id` INT(11) NOT NULL , `user_id` INT(11) NOT NULL , `comment` VARCHAR(255) NOT NULL , `comment_author` VARCHAR(255) NOT NULL , `date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`com_id`)) ENGINE = MyISAM;
