CREATE TABLE `chat`.`chat_message` ( `chat_message_id` INT(11) NOT NULL AUTO_INCREMENT , `to_user_id` INT(11) NOT NULL  , `from_user_id` INT(11) NOT NULL , `chat_message` TEXT NOT NULL , `timestamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `status` INT(1) NOT NULL , PRIMARY KEY (`chat_message_id`)) ENGINE = MyISAM;
